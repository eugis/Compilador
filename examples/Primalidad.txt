{

    n = read("Ingrese un n√∫mero:");

            primo = 1;
    }else {
            if ((n % 2) != 0 && ((n % 3) != 0)){
                i = 5;
                while(i < n && forced==0){
                    if ((n % i) == 0)
                        forced = 1;

}
